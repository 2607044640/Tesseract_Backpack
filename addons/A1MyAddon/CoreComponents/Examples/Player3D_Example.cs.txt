using Godot;
using Godot.Composition;

/// <summary>
/// Player3D 实体示例
/// 
/// 这是一个完整的玩家实体，展示了如何使用 Godot.Composition 框架。
/// 
/// 功能：
/// - WASD 移动
/// - 空格跳跃
/// - 鼠标控制相机
/// - 角色自动转向移动方向
/// - 动画自动切换
/// 
/// 使用的组件：
/// - PlayerInputComponent - 读取玩家输入
/// - MovementComponent - 处理物理移动
/// - CharacterRotationComponent - 控制角色朝向
/// - CameraControlComponent - 控制相机
/// - AnimationControllerComponent - 控制动画
/// </summary>
[Entity]
public partial class Player3D : CharacterBody3D
{
    /// <summary>
    /// Entity 的唯一职责：初始化组件系统
    /// Godot.Composition 会自动：
    /// 1. 收集所有子组件
    /// 2. 解析组件依赖关系
    /// 3. 注入依赖
    /// 4. 调用 OnEntityReady()
    /// </summary>
    public override void _Ready()
    {
        InitializeEntity();
        
        // 就这么简单！不需要任何其他代码。
        // 所有逻辑都在组件中。
    }
}

/* 
 * 场景结构：
 * 
 * Player3D (CharacterBody3D) - 本脚本
 * ├── PlayerInputComponent (Node)
 * ├── MovementComponent (Node)
 * ├── CharacterRotationComponent (Node)
 * ├── CameraControlComponent (Node)
 * ├── AnimationControllerComponent (Node)
 * ├── CollisionShape3D
 * ├── CharacterModel (Node3D)
 * │   └── AnimationPlayer
 * └── CameraPivot (Node3D)
 *     └── SpringArm3D
 *         └── Camera3D
 * 
 * 组件依赖关系：
 * 
 * PlayerInputComponent (独立)
 *   ↓ 事件
 * MovementComponent (依赖 PlayerInputComponent)
 *   ↓ 事件
 * CharacterRotationComponent (依赖 PlayerInputComponent)
 * 
 * CameraControlComponent (独立)
 * AnimationControllerComponent (读取 parent.Velocity)
 */
